Mixins, extends, helpers

.clearfix {
  &:before,
  &:after {
    content: '';
    display: table;
  }
  &:after {
    clear: both;
  }
}

.unstyle-list {
  list-style-type: none;
  margin: 0;
  padding: 0;
}

@mixin transition($length) {
  -webkit-transition: all $length ease-out;
     -moz-transition: all $length ease-out;
       -o-transition: all $length ease-out;
          transition: all $length ease-out;
}
